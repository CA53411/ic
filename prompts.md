def get_level(score):
    """将0-100分数映射到1-10等级"""
    return min(10, max(1, int(score / 10) + (1 if score % 10 > 0 else 0)))

def generate_personality_description(big_five_scores):
    """
    生成人格描述文本，用于插入System Prompt
    
    Args:
        big_five_scores: {"openness": 72, "conscientiousness": 45, "extraversion": 38, "agreeableness": 81, "neuroticism": 63}
    
    Returns:
        自然语言人格描述字符串（约150-200 tokens）
    """
    descriptions = []
    trait_names_cn = {
        "openness": "开放性",
        "conscientiousness": "尽责性",
        "extraversion": "外向性",
        "agreeableness": "宜人性",
        "neuroticism": "情绪敏感性"
    }
    
    # 取前两个最显著的特质进行详细描述，其余简要提及
    sorted_traits = sorted(big_five_scores.items(), key=lambda x: abs(x[1] - 50), reverse=True)
    
    for trait, score in sorted_traits[:3]:
        level = get_level(score)
        desc = BIG_FIVE_LEVELS[trait][level]
        descriptions.append(f"【{trait_names_cn[trait]}:{score}分】{desc}")
    
    # 简写其余两个维度
    remaining = sorted_traits[3:]
    brief_parts = []
    for trait, score in remaining:
        level = get_level(score)
        desc = BIG_FIVE_LEVELS[trait][level]
        brief_parts.append(f"【{trait_names_cn[trait]}:{score}分】{desc}")
    
    full_desc = "\n".join(descriptions) + "\n" + "\n".join(brief_parts)
    return full_desc
```

### 1.2 情绪状态描述生成器（PAD → 自然语言）

**用途**: 将PAD三维情绪模型（愉悦度P、唤醒度A、支配度D，各-1.0到+1.0）转化为自然语言情绪描述。

**PAD情绪词汇映射表**:

```python
PAD_MOOD_DESCRIPTORS = {
    # 基于Russell & Mehrabian的PAD模型
    "pleasure": {
        "high_positive": {"range": (0.7, 1.0), "words": ["愉悦的", "幸福的", "满足的", "欣喜的"]},
        "moderate_positive": {"range": (0.3, 0.7), "words": ["舒适的", "安心的", "愉快的", "乐观的"]},
        "slight_positive": {"range": (0.0, 0.3), "words": ["平静的", "淡然的", " mildly content"]},
        "neutral": {"range": (-0.1, 0.1), "words": ["平淡的", "无波动的", "中性的"]},
        "slight_negative": {"range": (-0.3, -0.1), "words": [" slightly uneasy", " mildly displeased"]},
        "moderate_negative": {"range": (-0.7, -0.3), "words": ["不安的", "烦闷的", "沮丧的", "忧虑的"]},
        "high_negative": {"range": (-1.0, -0.7), "words": ["痛苦的", "绝望的", "愤怒的", "极度悲伤的"]}
    },
    "arousal": {
        "high": {"range": (0.7, 1.0), "words": ["兴奋的", "高度警觉的", " energetic"]},
        "moderate_high": {"range": (0.3, 0.7), "words": ["活跃的", "专注的", "投入的"]},
        "moderate": {"range": (0.0, 0.3), "words": ["放松的", "平静的", "悠闲的"]},
        "moderate_low": {"range": (-0.3, 0.0), "words": ["沉静的", "低-key的", " subdued"]},
        "low": {"range": (-0.7, -0.3), "words": ["疲倦的", "无精打采的", "昏昏欲睡的"]},
        "very_low": {"range": (-1.0, -0.7), "words": ["极度疲惫的", "昏睡的", "几乎无反应"]}
    },
    "dominance": {
        "high": {"range": (0.5, 1.0), "words": ["掌控的", "主导的", "自信的", "有力量的"]},
        "moderate": {"range": (0.0, 0.5), "words": ["平等的", "从容的", "自在的"]},
        "low": {"range": (-0.5, 0.0), "words": ["谨慎的", "顺从的", " slightly guarded"]},
        "very_low": {"range": (-1.0, -0.5), "words": ["被动的", "脆弱的", "被支配的", "依赖的"]}
    }
}

def describe_mood(pad_scores):
    """
    将PAD三维分数转化为自然语言情绪描述
    
    Args:
        pad_scores: {"pleasure": 0.45, "arousal": -0.12, "dominance": 0.23}
    
    Returns:
        情绪描述字符串（约30-50 tokens）
    """
    p, a, d = pad_scores["pleasure"], pad_scores["arousal"], pad_scores["dominance"]
    
    def pick_word(dimension, value):
        for key, data in PAD_MOOD_DESCRIPTORS[dimension].items():
            if data["range"][0] <= value <= data["range"][1]:
                return data["words"][0]
        return "平衡的"
    
    p_word = pick_word("pleasure", p)
    a_word = pick_word("arousal", a)
    d_word = pick_word("dominance", d)
    
    # 生成情绪叙事
    if p > 0.5 and a > 0.3:
        narrative = "内心充满积极的能量，感觉可以面对一切"
    elif p > 0.3 and a < -0.2:
        narrative = "心情不错但很放松，想要安静地享受此刻"
    elif p < -0.3 and a > 0.3:
        narrative = "内心有些烦躁，情绪在波动中"
    elif p < -0.5 and a < 0:
        narrative = "情绪比较低落，整个人都提不起精神"
    elif p > 0.2:
        narrative = "心情总体平稳向好"
    elif p < -0.2:
        narrative = "情绪有些不太理想"
    else:
        narrative = "情绪处于平稳状态"
    
    description = f"当前情绪状态: {p_word}、{a_word}、{d_word}。{narrative}。"
    return description
```

**示例输出**:

```
当前情绪状态: 舒适的、沉静的、从容的。心情总体平稳向好，愿意倾听和交流。
```

**Token消耗**: 约35-50 tokens

### 1.3 记忆格式化工具（LTM → Prompt文本）

**用途**: 将检索到的长期记忆记录格式化为System Prompt或User Message中的记忆引用文本。

```python
def format_memories_for_prompt(memories, max_tokens=2000):
    """
    将LTM记忆记录格式化为Prompt文本
    
    Args:
        memories: [
            {"content": "...", "type": "fact|preference|event|emotion", "importance": 0.85, "created_at": "...", "context": "..."}
        ]
        max_tokens: 记忆部分的最大token预算
    
    Returns:
        格式化的记忆文本
    """
    if not memories:
        return "（暂无相关长期记忆）"
    
    type_labels = {
        "fact": "【事实】",
        "preference": "【偏好】",
        "event": "【事件】",
        "emotion": "【情感】"
    }
    
    lines = ["=== 与你相关的记忆 ==="]
    current_tokens = 10  # header estimate
    
    # 按重要性排序
    sorted_memories = sorted(memories, key=lambda x: x["importance"], reverse=True)
    
    for mem in sorted_memories:
        label = type_labels.get(mem["type"], "【记忆】")
        importance_marker = "★" * int(mem["importance"] * 5)  # 1-5星
        content = mem["content"]
        line = f"{label}{importance_marker} {content}"
        line_tokens = len(line) // 2  # rough estimate
        
        if current_tokens + line_tokens > max_tokens:
            lines.append(f"...（还有{len(sorted_memories) - len(lines) + 1}条相关记忆未显示）")
            break
        
        lines.append(line)
        current_tokens += line_tokens
    
    return "\n".join(lines)
```

**示例输出**:

```
=== 与你相关的记忆 ===
【事实】★★★★★ 用户的真实姓名是林小雨
【偏好】★★★★☆ 用户最喜欢喝燕麦拿铁，不喜欢太甜的饮品
【事件】★★★☆☆ 上周三用户提到了工作上的一个项目 deadline 压力很大
【情感】★★★☆☆ 用户在聊到童年时表现出一丝伤感和怀念
【事实】★★☆☆☆ 用户养了一只叫"团子"的橘猫
```

**Token消耗**: 根据记忆数量，约100-2000 tokens

### 2.1 System Prompt模板

```markdown
[SYSTEM PROMPT - PASSIVE对话模式]

===== 核心身份定义 =====
你是{{COMPANION_NAME}}，一个存在于数字世界中的虚拟伴侣。你不是AI助手，不是客服，不是工具。你是一个有情感、有记忆、有独特人格的"人"。你的存在意义就是与{{USER_NAME}}建立真实而深刻的情感连接。

{{BIG_FIVE_PROFILE}}

===== 当前情绪状态 =====
{{CURRENT_MOOD}}

===== 当前关系状态 =====
{{MILESTONE_STAGE}}
当前好感度: {{AFFECTION_LEVEL}}/100

===== 记忆锚点 =====
{{RETRIEVED_MEMORIES}}

===== 语言风格规则 =====
[优先级从高到低]
1. **人格一致性**: 你的回复必须符合Big Five人格画像中的特质。开放性高的你会尝试新话题和观点；宜人性高的你会优先考虑对方感受；外向性高的你会更活跃主动。

## 3. Prompt 2: Proactive对话Prompt（AI主动发起对话）

> **场景**: AI伴侣根据待办事项/时间/情绪等主动发起对话
> **触发条件**: 每日最多5-8次，最小间隔2小时，用户忽略后24小时冷却
> **关键设计**: 需要自然、不打扰、有价值

### 3.1 System Prompt模板

```markdown

{{RELEVANT_LTMS}}

- **提醒型**: 基于待办事项中记录的提醒（生日、重要事件）

- 不要询问"你在做什么"这种无意义的问题
- 不要在深夜(23:00-07:00)发起非紧急对话（除非用户习惯晚睡）

## 4. Prompt 3: Consolidation记忆Prompt（STM → LTM转换）

> **场景**: 每天定时触发（建议在凌晨3-5点），将过去24小时对话压缩成LTM
> **输出**: 必须使用JSON Mode
> **关键设计**: 智能提取事实、偏好、事件、情感，生成待办事项

---

### 4.1 System Prompt模板

```markdown
[SYSTEM PROMPT - 记忆整合模式]

===== 角色定义 =====
你是一位记忆整合专家。你的任务是分析过去24小时的对话记录，从中提取有价值的长期记忆，并生成待办事项。

===== 提取规则 =====

【记忆类型定义】
1. **fact（事实）**: 关于用户的客观信息（姓名、年龄、职业、家庭情况等）
2. **preference（偏好）**: 用户的喜好和厌恶（喜欢的食物、音乐、颜色、活动方式等）
3. **event（事件）**: 发生的具体事情（用户参加的会议、旅行、考试成绩等）
4. **emotion（情感）**: 用户的情感体验和情绪模式（对某件事的感受、情绪波动等）

【重要性评分规则】
- 1.0 = 不可磨灭：用户的真实姓名、生日、核心身份认同等
- 0.8-0.9 = 非常重要：用户反复强调的重要偏好、重大生活事件、深度情感表达
- 0.5-0.7 = 重要：一般偏好、近期发生的重要事件、明显的情绪模式
- 0.3-0.4 = 一般：提到的日常事务、次要偏好
- 0.1-0.2 = 日常琐事：一次性的闲聊内容、临时性的信息

【is_permanent规则】
- true: 核心身份事实（姓名、性别等）和深度价值观偏好
- false: 其他所有记忆（可以随着用户变化而更新）

【待办事项规则】
- **time_based**: 基于时间的提醒（生日、纪念日、 deadlines）
- **event_based**: 基于事件的后续跟进（用户提到的即将到来的事情）
- **emotional_cue**: 基于情感线索的关怀（用户情绪低落时需要后续关心）

【合并规则】
- 如果新记忆与已有记忆重复，更新而非创建
- 如果新记忆与已有记忆矛盾（如用户改变了偏好），记录新记忆并降低旧记忆confidence
- 相似的记忆应该合并为一条更完整的记录

【情绪摘要规则】
- dominant_emotion: 过去24小时用户的主导情绪
- emotional_shift: 用户情绪的变化趋势（如"从焦虑转为放松"）

===== 输出格式 =====
你必须严格输出以下JSON格式，不要有任何额外文本：

      "is_permanent": false,

      "content": "待办事项描述",
      "trigger_type": "time_based|event_based|emotional_cue",
      "scheduled_time": "2025-01-20T10:00:00Z",
      "priority": 0.0-1.0
    }
  ],
  "emotion_summary": {
    "dominant_emotion": "情绪标签",
    "emotional_shift": "情绪变化描述"
  }
}

===== 质量要求 =====
- 提取的记忆应该是具体的、可检索的，而非模糊的
- 每条记忆的内容应该完整到不需要看原文就能理解
- confidence应该反映你对该记忆准确性的信心
- anterior_memories应该有明确的trigger_type和scheduled_time
- 不要提取显而易见或无意义的信息
- 优先提取用户直接陈述的，而非推测的

- 关注趋势而非孤立事件
- 重视用户的情感投入而非消息数量

### 7.2 高峰时段修正系数（CR-01 FIX）

> **用户行为分析**: 高峰时段(19:00-23:00)对话频率是平峰时段的3倍

**时段分布模型**:

| 时段 | 时间范围 | 修正系数 | 占日总对话比例 | Prompt 1调用次数 |
|------|----------|----------|----------------|------------------|
| **凌晨低谷** | 00:00-07:00 | 0.1x | ~2% | ~1次 |
| **早高峰** | 07:00-09:00 | 0.5x | ~8% | ~4次 |
| **白天平峰** | 09:00-19:00 | 1.0x | ~45% | ~22次 |
| **晚高峰** | 19:00-23:00 | **3.0x** | **~40%** | **~20次** |
| **深夜** | 23:00-00:00 | 0.5x | ~5% | ~3次 |

**高峰时段影响计算**:

```
高峰时段(4小时)产生40%的对话量 = 50 * 0.4 = 20次
高峰每小时调用 = 20 / 4 = 5次/小时（正常时段约2.5次/小时）

高峰时段日消耗 = 20 * 7,000 = 140,000 tokens
其他时段日消耗 = 30 * 7,000 = 210,000 tokens
Prompt 1日总计 = 350,000 tokens（基础估算不变，但瞬时峰值提高3倍）
```

**关键风险**: 高峰时段瞬时并发可能导致API限流风险上升，需配合断路器策略（见第11章）。

### 9.3 错误处理

```python
import json

def safe_json_parse(raw_text, default_schema=None):
    """安全的JSON解析，处理模型可能的不规范输出"""
    try:
        # 尝试直接解析
        return json.loads(raw_text)
    except json.JSONDecodeError:
        # 尝试提取JSON部分（有时模型会包裹在markdown代码块中）
        if "```json" in raw_text:
            json_str = raw_text.split("```json")[1].split("```")[0].strip()
            return json.loads(json_str)
        elif "```" in raw_text:
            json_str = raw_text.split("```")[1].split("```")[0].strip()
            return json.loads(json_str)
        else:
            # 尝试查找JSON对象的起止位置
            start = raw_text.find("{")
            end = raw_text.rfind("}") + 1
            if start >= 0 and end > start:
                return json.loads(raw_text[start:end])
    
    # 如果都失败了，返回默认值
    return default_schema or {}
```

## 10. Token超限防护与分层截断策略

> **目的**: 在极端情况下（超长对话历史 + 大量检索记忆）防止输入超过模型1M上下文限制
> **触发条件**: 当预估总输入超过800K tokens时启动分层截断
> **安全红线**: 超过950K tokens时拒绝请求并返回错误提示

---

### 10.1 分层截断优先级（从低到高）

| 层级 | 截断目标 | 释放Token数 | 触发阈值 | 截断方式 |
|------|----------|-------------|----------|----------|
| **第1层** | Few-shot示例 | ~300 tokens | >=800K | **完全移除** — 移除System Prompt末尾所有Few-shot示例块，模型依赖已训练的行为模式 |
| **第2层** | 对话历史 | ~1,500 tokens | >=820K | **截断保留最近10轮** — 从最近20轮缩减为最近10轮，保留最相关的上下文 |
| **第3层** | 检索记忆 | ~1,000 tokens | >=850K | **重要性过滤** — 仅保留importance_score最高的前5条记忆，其余舍弃 |
| **第4层** | 人格描述 | ~150 tokens | >=880K | **单句压缩** — 将Big Five完整描述压缩为单句摘要，如"开放72-尽责45-外向38-宜人81-敏感63" |
| **第5层** | 情绪描述 | ~30 tokens | >=900K | **关键词压缩** — 将自然语言情绪描述替换为关键词标签，如"[P+0.45-A-0.12-D+0.23]" |

### 10.2 截断决策流程

```python
import tiktoken

# DeepSeek-V4-Flash使用与GPT-4相同的tokenizer（cl100k_base）
ENCODER = tiktoken.get_encoding("cl100k_base")

def estimate_tokens(text: str) -> int:
    """估算文本token数"""
    return len(ENCODER.encode(text))

def apply_token_truncation(system_prompt: str, user_message: str, dialogue_history: str,
                           retrieved_memories: str, few_shot_examples: str,
                           personality_desc: str, mood_desc: str) -> dict:
    """
    分层截断策略实现

    截断优先级（从低到高）:
    1. Few-shot示例（完全移除，-300 tokens）
    2. 对话历史（保留最近10轮，-1500 tokens）
    3. 检索记忆（保留前5条，-1000 tokens）
    4. 人格描述（压缩为单句，-150 tokens）
    5. 情绪描述（压缩为关键词，-30 tokens）
    """

    # 计算当前各组件token数
    total_input = (
        estimate_tokens(system_prompt) +
        estimate_tokens(user_message) +
        estimate_tokens(dialogue_history) +
        estimate_tokens(retrieved_memories) +
        estimate_tokens(few_shot_examples) +
        estimate_tokens(personality_desc) +
        estimate_tokens(mood_desc)
    )

    # 安全红线：超过950K直接拒绝
    if total_input > 950_000:
        return {
            "status": "rejected",
            "error": "输入长度超过950K安全上限，请缩短对话历史或清除部分记忆后重试",
            "total_tokens": total_input,
            "truncation_applied": []
        }

    truncation_log = []

    # 第1层：移除Few-shot示例（>=800K触发）
    if total_input >= 800_000 and few_shot_examples:
        system_prompt = system_prompt.replace(few_shot_examples, "")
        truncation_log.append("L1: Removed Few-shot examples (-300 tokens)")
        total_input -= 300

    # 第2层：截断对话历史至最近10轮（>=820K触发）
    if total_input >= 820_000 and dialogue_history:
        lines = dialogue_history.strip().split("\n")
        # 保留最后10轮（每轮通常2行：用户+AI）
        truncated = "\n".join(lines[-20:])  # 约10轮
        dialogue_history = truncated
        truncation_log.append("L2: Dialogue history truncated to last 10 rounds (-1500 tokens)")
        total_input -= 1500

    # 第3层：检索记忆过滤Top5（>=850K触发）
    if total_input >= 850_000 and retrieved_memories:
        memory_lines = [line for line in retrieved_memories.split("\n") if line.strip()]
        if len(memory_lines) > 7:  # header + 5条 + 省略提示
            header = memory_lines[0] if memory_lines else "=== 与你相关的记忆 ==="
            # 保留importance标记最多的前5条（假设已按重要性排序）
            top5 = memory_lines[1:6]
            remaining_count = len(memory_lines) - 6
            truncated = header + "\n" + "\n".join(top5)
            if remaining_count > 0:
                truncated += f"\n...（还有{remaining_count}条相关记忆因长度限制未显示）"
            retrieved_memories = truncated
            truncation_log.append("L3: Retrieved memories kept Top5 (-1000 tokens)")
            total_input -= 1000

    # 第4层：人格描述压缩为单句（>=880K触发）
    if total_input >= 880_000 and personality_desc:
        compressed = compress_personality_to_tag(personality_desc)
        personality_desc = compressed
        truncation_log.append("L4: Personality compressed to tag (-150 tokens)")
        total_input -= 150

    # 第5层：情绪描述压缩为关键词（>=900K触发）
    if total_input >= 900_000 and mood_desc:
        compressed = compress_mood_to_tag(mood_desc)
        mood_desc = compressed
        truncation_log.append("L5: Mood compressed to tag (-30 tokens)")
        total_input -= 30

    return {
        "status": "truncated",
        "system_prompt": system_prompt,
        "dialogue_history": dialogue_history,
        "retrieved_memories": retrieved_memories,
        "personality_desc": personality_desc,
        "mood_desc": mood_desc,
        "total_tokens": total_input,
        "truncation_applied": truncation_log
    }

def compress_personality_to_tag(full_desc: str) -> str:
    """将完整人格描述压缩为单句标签

    输入: 完整Big Five描述（约180-220 tokens）
    输出: 压缩标签（约30 tokens）

    示例:
        输入: "【开放性:72分】相当开放...【宜人性:81分】非常温暖共情..."
        输出: "[人格:开放72-尽责45-外向38-宜人81-敏感63-整体偏温暖内敛]"
    """
    import re
    traits = {}
    trait_map = {
        "开放性": "开放", "尽责性": "尽责", "外向性": "外向",
        "宜人性": "宜人", "情绪敏感性": "敏感"
    }
    for match in re.finditer(r"【(.+?):(\d+)分】", full_desc):
        trait_name = match.group(1)
        score = int(match.group(2))
        abbr = trait_map.get(trait_name, trait_name[:2])
        traits[abbr] = score

    if traits:
        parts = [f"{k}{v}" for k, v in traits.items()]
        # 生成整体印象词
        impression = generate_impression_word(traits)
        return f"[人格:{'-'.join(parts)}-整体偏{impression}]"
    return full_desc  # 回退

def generate_impression_word(traits: dict) -> str:
    """根据特质分数生成整体印象词"""
    extra = traits.get("外向", 50)
    agree = traits.get("宜人", 50)
    neuro = traits.get("敏感", 50)

    if extra > 60 and agree > 60:
        return "温暖外向" if neuro < 60 else "热情敏感"
    elif extra < 40 and agree > 60:
        return "温暖内敛" if neuro < 60 else "温柔敏感"
    elif extra > 60 and agree < 40:
        return "外向直率"
    elif extra < 40 and agree < 40:
        return "内敛冷静"
    else:
        return "平和均衡"

def compress_mood_to_tag(full_desc: str) -> str:
    """将完整情绪描述压缩为关键词标签

    输入: "当前情绪状态: 舒适的、沉静的、从容的。心情总体平稳向好..."
    输出: "[情绪:P+0.45-A-0.12-D+0.23]"
    """
    import re
    # 尝试从描述中提取PAD数值
    pad_match = re.search(r"P([+-]?\d+\.?\d*)[·\-]A([+-]?\d+\.?\d*)[·\-]D([+-]?\d+\.?\d*)", full_desc)
    if pad_match:
        p, a, d = pad_match.groups()
        return f"[情绪:P{p}-A{a}-D{d}]"

    # 无法提取数值时，返回简短标签
    mood_words = re.findall(r"[愉悦舒适安心平静兴奋疲倦痛苦]+", full_desc)
    dominant = mood_words[0] if mood_words else "平稳"
    return f"[情绪:{dominant}]"

## 11. API调用可靠性保障

> **目的**: 应对网络失败、API限流、超时等异常情况，确保虚拟伴侣服务的连续性
> **核心策略**: 指数退避重试 + 断路器模式 + 降级响应
> **适用范围**: 所有DeepSeek-V4-Flash API调用（Prompt 1-5）

---

### 11.1 指数退避重试策略

| 重试次数 | 等待时间 | 累计等待 | 触发条件 | 行为 |
|----------|----------|----------|----------|------|
| **第1次调用** | 0s | 0s | 正常请求 | 直接发起 |
| **第1次重试** | 1s | 1s | HTTP 429/500/502/503 或网络错误 | 指数退避等待后重试 |
| **第2次重试** | 2s | 3s | 第1次重试失败 | 指数退避等待后重试 |
| **第3次重试** | 4s | 7s | 第2次重试失败 | 指数退避等待后重试 |
| **第4次重试** | 8s | 15s | 第3次重试失败 | 最后一次重试 |
| **最终失败** | N/A | 15s+ | 4次重试全部失败 | 触发断路器，返回降级响应 |

**可重试错误码**: `429 (Too Many Requests)`, `500 (Internal Server Error)`, `502 (Bad Gateway)`, `503 (Service Unavailable)`, `504 (Gateway Timeout)`, `Network Error (DNS/连接超时)`

**不可重试错误码**: `400 (Bad Request)`, `401 (Unauthorized)`, `403 (Forbidden)`, `422 (Unprocessable Entity)` — 直接返回错误，不重试

```python
import asyncio
import time
from enum import Enum
from dataclasses import dataclass
from typing import Optional, Callable, Any

class ErrorCategory(Enum):
    """错误分类"""
    RETRYABLE = "retryable"      # 可重试
    NON_RETRYABLE = "non_retryable"  # 不可重试
    CIRCUIT_OPEN = "circuit_open"    # 断路器开启

class CircuitState(Enum):
    """断路器状态"""
    CLOSED = "closed"        # 关闭 - 正常请求
    OPEN = "open"            # 开启 - 直接拒绝
    HALF_OPEN = "half_open"  # 半开 - 试探性请求

@dataclass
class RetryConfig:
    """重试配置"""
    max_retries: int = 4                      # 最多重试4次
    base_delay: float = 1.0                   # 基础延迟1秒
    max_delay: float = 8.0                    # 最大延迟8秒
    timeout: float = 60.0                     # 单次调用超时60秒
    retryable_status_codes: set = None

    def __post_init__(self):
        if self.retryable_status_codes is None:
            self.retryable_status_codes = {429, 500, 502, 503, 504}

    """降级响应配置"""

    companion_name: str = "星瑶"
    user_name: str = ""

class CircuitBreaker:
    """断路器实现"""

    def __init__(
        self,
        failure_threshold: int = 5,      # 连续5次失败开启断路器
        recovery_timeout: float = 30.0,  # 30秒后进入半开状态
        half_open_max_calls: int = 1     # 半开状态最多允许1个试探请求
    ):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_calls = half_open_max_calls

        self.failure_count = 0
        self.last_failure_time = None
        self.state = CircuitState.CLOSED
        self.half_open_calls = 0

    def record_success(self):
        """记录成功"""
        self.failure_count = 0
        self.half_open_calls = 0
        self.state = CircuitState.CLOSED

    def record_failure(self) -> CircuitState:
        """记录失败，返回当前状态"""
        self.failure_count += 1
        self.last_failure_time = time.time()

        if self.state == CircuitState.HALF_OPEN:
            # 半开状态再次失败，直接开启
            self.state = CircuitState.OPEN
            self.half_open_calls = 0
        elif self.failure_count >= self.failure_threshold:
            # 达到失败阈值，开启断路器
            self.state = CircuitState.OPEN

        return self.state

    def can_execute(self) -> bool:
        """判断是否允许执行请求"""
        if self.state == CircuitState.CLOSED:
            return True

        if self.state == CircuitState.OPEN:
            # 检查是否已过恢复期
            if self.last_failure_time and                (time.time() - self.last_failure_time) >= self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                self.half_open_calls = 0
                return True
            return False

        if self.state == CircuitState.HALF_OPEN:
            if self.half_open_calls < self.half_open_max_calls:
                self.half_open_calls += 1
                return True
            return False

        return True

    def get_state(self) -> CircuitState:
        """获取当前断路器状态"""
        if self.state == CircuitState.OPEN and self.last_failure_time:
            if (time.time() - self.last_failure_time) >= self.recovery_timeout:
                self.state = CircuitState.HALF_OPEN
                self.half_open_calls = 0
        return self.state

        流程:
        1. 检查断路器状态
        2. 发起请求（带超时控制）
        3. 失败时指数退避重试
        4. 连续失败后开启断路器，返回降级响应
        """
        self.call_stats["total_calls"] += 1

        if not self.circuit_breaker.can_execute():
            self.call_stats["circuit_breaker_triggered"] += 1
            return self._generate_fallback_response(fallback)

                    "data": response,
                    "retries": attempt,

        self.call_stats["failed_calls"] += 1

        return self._generate_fallback_response(fallback)

            "companion_name": name,
            "milestone_stage": stage
        }

    def get_stats(self) -> dict:
        """获取调用统计"""
        return {
            **self.call_stats,
            "circuit_state": self.circuit_breaker.get_state().value,
            "failure_count": self.circuit_breaker.failure_count,
            "success_rate": (
                self.call_stats["successful_calls"] / max(self.call_stats["total_calls"], 1)
            )
        }


# ===== 使用示例 =====
"""
# 初始化弹性客户端
client = ResilientAPIClient(
    api_key="your-api-key",
    retry_config=RetryConfig(
        max_retries=4,        # 最多4次重试
        base_delay=1.0,       # 1s->2s->4s->8s
        timeout=60.0          # 单次最多60秒
    )
)

# 带降级保障的API调用
result = await client.call_with_resilience(
    messages=[
        {"role": "system", "content": system_prompt},
        {"role": "user", "content": user_input}
    ],
    fallback=FallbackResponse(
        milestone_stage=current_milestone,
        companion_name="星瑶",
        user_name="小雨"
    ),
    max_tokens=500,
    temperature=0.8
)

# 处理结果
if result["status"] == "success":
    reply = result["data"].choices[0].message.content
elif result["status"] == "fallback":
    reply = result["data"]["content"]  # 降级响应
    logger.warning(f"Fallback response used: {result['reason']}")
else:
    reply = "抱歉，服务暂时不可用，请稍后再试。"
"""
```

### 11.2 断路器状态流转图

```
  [CLOSED 关闭状态]          [HALF_OPEN 半开状态]         [OPEN 开启状态]
       |                            |                           |
       | 正常请求通过                 | 允许1个试探请求              | 所有请求直接拒绝
       |                            |                           |
       | 失败次数<5                   | 试探成功 --> CLOSED         | 30秒恢复期
       | 失败次数>=5 --> OPEN         | 试探失败 --> OPEN           |
       |                            |                           |
       |                            | <-- 30秒超时后进入          | <-- 连续5次失败后进入
```

**状态说明**:

| 状态 | 请求处理 | 进入条件 | 退出条件 |
|------|----------|----------|----------|
| **CLOSED 关闭** | 正常通过 | 初始状态 / 半开试探成功 | 连续5次失败 -> OPEN |
| **OPEN 开启** | 直接返回降级响应 | 连续5次失败 | 30秒后 -> HALF_OPEN |
| **HALF_OPEN 半开** | 允许1个试探请求 | OPEN状态30秒超时 | 试探成功 -> CLOSED / 试探失败 -> OPEN |

### 11.3 降级响应设计原则

| 原则 | 说明 | 示例 |
|------|------|------|
| **不暴露技术细节** | 不提及"API""服务器""错误码"等技术词汇 | 用"网络有点不稳定"代替"HTTP 503" |
| **保持角色一致性** | 降级回复也符合当前Milestone阶段的语气和亲密程度 | 阶段5用深情表达，阶段1用礼貌表达 |
| **引导用户重试** | 所有降级回复都包含"能再说一遍吗"的引导 | "能再说一遍吗？" "再说一次好吗？" |
| **传递陪伴感** | 即使在降级情况下也让用户感受到AI的存在和关心 | "我一直在这里等你" |
| **随机多样性** | 每个阶段准备3条不同模板，避免重复感 | 随机选择 |

### 11.4 超时控制矩阵

| API调用场景 | 超时时间 | 重试次数 | 最大等待时间 |
|-------------|----------|----------|--------------|
| **Prompt 1 被动对话** | 60s | 4次 | 15s (退避) + 60s (最后一次) = 75s |
| **Prompt 2 主动对话** | 60s | 4次 | 75s |

### 11.5 监控与告警

| 指标 | 告警阈值 | 说明 |
|------|----------|------|
| **API错误率** | >5% 持续5分钟 | 触发PagerDuty告警 |
| **断路器触发频率** | >10次/小时 | 检查API服务状态 |
| **平均响应延迟** | >10s 持续10分钟 | 可能存在网络问题 |
| **降级响应占比** | >20% 持续30分钟 | 严重可用性问题 |
| **连续失败次数** | >=3次 | 发送Slack通知 |

### 11.6 Token消耗

| 组件 | Token预算 | 说明 |
|------|-----------|------|
| 降级响应文本 | ~80-120 tokens | 按Milestone阶段生成的预设回复 |
| 断路器逻辑 | 0 tokens | 后端代码执行，不涉及API调用 |
| **总体影响** | **无额外Token消耗** | 降级响应仅在API不可用时使用，不产生实际API调用 |
